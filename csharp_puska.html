9 Teljes megoldás (mai_menu vizsgafeladat)

Itt egyben megtalálod a feladat leegyszerűsített, jól felépített megoldását, ami tökéletes minta a hasonló vizsgafeladatokhoz.

Konzolos alkalmazás (Program.cs) 
using System.Text;

namespace mai_menu;

class Hozzavalo
{
    public string Mennyiseg = "";
    public string Egyseg = "";
    public string Nev = "";
}

class Etel
{
    public string Nev = "";
    public string FajlNev = "";
    public List<Hozzavalo> Hozzavalok = new List<Hozzavalo>();
    public string Elkeszites = "";
}

class Program
{
    static List<Etel> etelek = new List<Etel>();
    static string mappa = AppDomain.CurrentDomain.BaseDirectory;

    static void Main()
    {
        BeolvasMenu("menu.txt");

        Console.WriteLine("Beolvasva " + etelek.Count + " foetel:");
        foreach (Etel e in etelek)
            Console.WriteLine("  - " + e.Nev + " (" + e.Hozzavalok.Count + " hozzavalo)");

        Console.WriteLine();
        KerdezQ1();

        Console.WriteLine();
        KerdezQ2();

        Console.WriteLine();
        KerdezQ3();

        Console.WriteLine();
        LetrehozKellTxt();

        Console.WriteLine();
        Console.ReadLine();
    }

    static void BeolvasMenu(string fajlnev)
    {
        string[] sorok = File.ReadAllLines(Path.Combine(mappa, fajlnev), Encoding.UTF8);

        foreach (string sor in sorok)
        {
            if (!sor.Contains(".txt"))
                continue;

            string[] szavak = sor.Trim().Split(new char[] { '\t', ' ' }, StringSplitOptions.RemoveEmptyEntries);

            string fajlNev = "";
            int fajlIndex = -1;

            for (int i = szavak.Length - 1; i >= 0; i--)
            {
                if (szavak[i].EndsWith(".txt"))
                {
                    fajlNev = szavak[i];
                    fajlIndex = i;
                    break;
                }
            }

            if (fajlNev == "")
                continue;

            string etelNev = "";
            for (int i = 0; i < fajlIndex; i++)
            {
                if (i > 0) etelNev = etelNev + " ";
                etelNev = etelNev + szavak[i];
            }

            if (etelNev.ToLower().Contains("leves"))
                continue;

            Etel uj = new Etel();
            uj.Nev = etelNev;
            uj.FajlNev = fajlNev;
            BeolvasHozzavalok(uj);
            etelek.Add(uj);
        }
    }

    static void BeolvasHozzavalok(Etel etel)
    {
        string utvonal = Path.Combine(mappa, etel.FajlNev);
        if (!File.Exists(utvonal)) return;

        string[] sorok = File.ReadAllLines(utvonal, Encoding.UTF8);
        bool hozzavalokResz = false;
        bool elkeszitesResz = false;
        string elkeszites = "";

        foreach (string sor in sorok)
        {
            if (sor.Trim().Contains("Hozzávalók:"))
            {
                hozzavalokResz = true;
                elkeszitesResz = false;
                continue;
            }

            if (sor.Trim().Contains("Elkészítés:"))
            {
                hozzavalokResz = false;
                elkeszitesResz = true;
                continue;
            }

            if (hozzavalokResz && sor.Trim() != "")
            {
                string[] reszek = sor.Trim().Split(new char[] { '\t', ' ' }, StringSplitOptions.RemoveEmptyEntries);

                if (reszek.Length >= 3)
                {
                    Hozzavalo h = new Hozzavalo();
                    h.Mennyiseg = reszek[0];
                    h.Egyseg = reszek[1];

                    string nev = "";
                    for (int i = 2; i < reszek.Length; i++)
                    {
                        if (i > 2) nev = nev + " ";
                        nev = nev + reszek[i];
                    }
                    h.Nev = nev;

                    etel.Hozzavalok.Add(h);
                }
            }

            if (elkeszitesResz && sor.Trim() != "")
                elkeszites = elkeszites + sor + "\n";
        }

        etel.Elkeszites = elkeszites;
    }

    static void KerdezQ1()
    {
        int osszes = 0;
        foreach (Etel e in etelek)
            osszes = osszes + e.Hozzavalok.Count;

        Console.WriteLine("1. Osszesen hany hozzavalo kell?");
        Console.WriteLine("   " + osszes + " hozzavalo szukseges.");
    }

    static void KerdezQ2()
    {
        Console.WriteLine("2. Melyik etelekhez kell egy adott hozzavalo?");
        Console.Write("   Keresett hozzavalo: ");
        string beker = Console.ReadLine();
        if (beker == null) beker = "";
        string keresett = beker.ToLower().Trim();

        List<string> talalatok = new List<string>();

        foreach (Etel e in etelek)
        {
            foreach (Hozzavalo h in e.Hozzavalok)
            {
                if (h.Nev.ToLower().Contains(keresett))
                {
                    talalatok.Add(e.Nev);
                    break;
                }
            }
        }

        if (talalatok.Count == 0)
        {
            Console.WriteLine("   Egyetlen etelhez sem kell.");
        }
        else
        {
            Console.Write("   Szukseges: ");
            for (int i = 0; i < talalatok.Count; i++)
            {
                if (i > 0) Console.Write(", ");
                Console.Write(talalatok[i]);
            }
            Console.WriteLine();
        }
    }

    static void KerdezQ3()
    {
        Console.WriteLine("3. Melyik hozzavalo szerepel mindharom etelben?");

        List<string> mindharomban = new List<string>();

        foreach (Hozzavalo h in etelek[0].Hozzavalok)
        {
            bool mindenbenVan = true;

            for (int i = 1; i < etelek.Count; i++)
            {
                bool benneVan = false;
                foreach (Hozzavalo h2 in etelek[i].Hozzavalok)
                {
                    if (h2.Nev.ToLower() == h.Nev.ToLower())
                    {
                        benneVan = true;
                        break;
                    }
                }

                if (!benneVan)
                {
                    mindenbenVan = false;
                    break;
                }
            }

            if (mindenbenVan)
                mindharomban.Add(h.Nev);
        }

        if (mindharomban.Count == 0)
            Console.WriteLine("   Nincs kozos hozzavalo mindharom etelben.");
        else
        {
            foreach (string nev in mindharomban)
                Console.WriteLine("   Mindharomban: " + nev);
        }
    }

    static void LetrehozKellTxt()
    {
        List<string> kulcsok = new List<string>();
        List<string> nevek = new List<string>();
        List<string> egysegek = new List<string>();
        List<double> mennyisegek = new List<double>();

        foreach (Etel e in etelek)
        {
            foreach (Hozzavalo h in e.Hozzavalok)
            {
                string kulcs = h.Nev.ToLower() + "|" + h.Egyseg.ToLower();

                double menny = 0;
                double.TryParse(h.Mennyiseg.Replace(",", "."), out menny);

                int index = kulcsok.IndexOf(kulcs);
                if (index == -1)
                {
                    kulcsok.Add(kulcs);
                    nevek.Add(h.Nev);
                    egysegek.Add(h.Egyseg);
                    mennyisegek.Add(menny);
                }
                else
                {
                    mennyisegek[index] = mennyisegek[index] + menny;
                }
            }
        }

        StreamWriter sw = new StreamWriter(Path.Combine(mappa, "kell.txt"), false, Encoding.UTF8);

        for (int i = 0; i < kulcsok.Count; i++)
            sw.WriteLine(nevek[i] + "\t" + egysegek[i] + "\t" + mennyisegek[i]);

        sw.Close();

        Console.WriteLine("kell.txt letrehozva (" + kulcsok.Count + " hozzavalo).");
    }
}

    private string GeneralFajlNev(string etelNev)
    {
        string s = etelNev;
        s = s.Replace("á", "a").Replace("Á", "A");
        s = s.Replace("é", "e").Replace("É", "E");
        s = s.Replace("í", "i").Replace("Í", "I");
        s = s.Replace("ó", "o").Replace("Ó", "O");
        s = s.Replace("ö", "o").Replace("Ö", "O");
        s = s.Replace("ő", "o").Replace("Ő", "O");
        s = s.Replace("ú", "u").Replace("Ú", "U");
        s = s.Replace("ü", "u").Replace("Ü", "U");
        s = s.Replace("ű", "u").Replace("Ű", "U");
        s = s.Replace(" ", "_");
        return s + ".txt";
    }

    // Ellenorzes gomb: fajl letezik-e? Feluliras?
    private void BtnEllenorzes_Click(object sender, EventArgs e)
    {
        if (txtNev.Text.Trim() == "")
        {
            MessageBox.Show("Az etel neve nem lehet ures!", "Hiba",
                MessageBoxButtons.OK, MessageBoxIcon.Warning);
            return;
        }

        string fajlNev = GeneralFajlNev(txtNev.Text.Trim());
        string utvonal = Path.Combine(mappa, fajlNev);

        if (File.Exists(utvonal))
        {
            DialogResult valasz = MessageBox.Show(
                "A '" + fajlNev + "' fajl mar letezik. Felulirja?",
                "Fajl mar letezik",
                MessageBoxButtons.YesNo,
                MessageBoxIcon.Question);

            if (valasz == DialogResult.No)
                return;
        }

        // Engedelyezzuk a tabokat
        tabControl.Enabled = true;
        ellenorzesOk = true;

        MessageBox.Show("Rendben! Toltse ki a hozzavalokat es az elkeszitesi modot.",
            "OK", MessageBoxButtons.OK, MessageBoxIcon.Information);
    }

    // Uj sor hozzaadasa a tablazathoz
    private void BtnHozzaad_Click(object sender, EventArgs e)
    {
        dgvHozzavalok.Rows.Add("", "", "");
    }

    // Kivalasztott sor torlese
    private void BtnTorol_Click(object sender, EventArgs e)
    {
        if (dgvHozzavalok.SelectedRows.Count > 0)
        {
            foreach (DataGridViewRow sor in dgvHozzavalok.SelectedRows)
            {
                if (!sor.IsNewRow)
                    dgvHozzavalok.Rows.Remove(sor);
            }
        }
    }

    // Mentes gomb: ellenorzesek + fajlba iras
    private void BtnMentes_Click(object sender, EventArgs e)
    {
        if (!ellenorzesOk)
        {
            MessageBox.Show("Elobb kattintson az Ellenorzes gombra!",
                "Figyelem", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            return;
        }

        // Hozzavalok osszegyujtese a tablazatbol
        List<Hozzavalo> hozzavalok = new List<Hozzavalo>();

        for (int i = 0; i < dgvHozzavalok.Rows.Count; i++)
        {
            DataGridViewRow sor = dgvHozzavalok.Rows[i];
            string menny = "";
            string egys = "";
            string nev = "";

            if (sor.Cells[0].Value != null) menny = sor.Cells[0].Value.ToString().Trim();
            if (sor.Cells[1].Value != null) egys = sor.Cells[1].Value.ToString().Trim();
            if (sor.Cells[2].Value != null) nev = sor.Cells[2].Value.ToString().Trim();

            // Csak a teljesen kitoltott sorokat vesszuk figyelembe
            if (menny != "" && egys != "" && nev != "")
            {
                Hozzavalo h = new Hozzavalo();
                h.Mennyiseg = menny;
                h.Egyseg = egys;
                h.Nev = nev;
                hozzavalok.Add(h);
            }
        }

        // Ellenorzes: van-e hozzavalo?
        if (hozzavalok.Count == 0)
        {
            MessageBox.Show("Legalabb egy hozzavalot meg kell adni!",
                "Hiany", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            return;
        }

        // Ellenorzes: van-e elkeszitesi mod?
        if (txtElkeszites.Text.Trim() == "")
        {
            MessageBox.Show("Az elkeszitesi modot meg kell adni!",
                "Hiany", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            return;
        }

        // Fajlba iras - ugyanolyan szerkezettel mint a tobbi etel fajl
        string fajlNev = GeneralFajlNev(txtNev.Text.Trim());
        string utvonal = Path.Combine(mappa, fajlNev);

        StreamWriter sw = new StreamWriter(utvonal, false, Encoding.UTF8);
        sw.WriteLine("Hozzávalók:");
        foreach (Hozzavalo h in hozzavalok)
        {
            sw.WriteLine(h.Mennyiseg + "\t" + h.Egyseg + "\t" + h.Nev);
        }
        sw.WriteLine("Elkészítés:");
        sw.WriteLine(txtElkeszites.Text.Trim());
        sw.Close();

        // Ellenorzes: sikerult-e a mentes?
        if (File.Exists(utvonal) && new FileInfo(utvonal).Length > 0)
        {
            MessageBox.Show("Sikeresen mentve: " + fajlNev,
                "Siker", MessageBoxButtons.OK, MessageBoxIcon.Information);
        }
        else
        {
            MessageBox.Show("Hiba a mentes soran!",
                "Hiba", MessageBoxButtons.OK, MessageBoxIcon.Error);
        }
    }

    // Belepo pont
    [STAThread]
    static void Main()
    {
        Application.EnableVisualStyles();
        Application.Run(new Form1());
    }

10 Vizsga GUI összerakása (Visual Studio Designer)

Pontos lépéssor, hogyan rakd össze a felületet kódolás nélkül, hogy utána csak a logikát kelljen beilleszteni (lásd a 9. pontot).

1. Előkészületek és a Form beállítása

  • Hozz létre egy új Windows Forms App (.NET) projektet.
  • Kattints rá a Form üres területére a Designerben.
  • A jobb oldali Properties (Tulajdonságok) ablakban állítsd be:
    • Text: "Új étel felvétele" (ez lesz az ablak címe)
    • Size: 620; 520
    • StartPosition: CenterScreen

2. A felső rész: Név és Ellenőrzés gomb

  • Húzz be egy Label-t bal felülre. Text-jét állítsd át erre: "Étel neve:".
  • Húzz mellé egy TextBox-ot. A (Name) tulajdonságát nevezd át erre: txtNev.
  • Húzz be mellé jobb oldalra egy Button-t.
    • (Name): btnEllenorzes
    • Text: "Ellenőrzés"
    • Kattints rá duplán a gombra! Ez létrehozza a kódban a btnEllenorzes_Click eseményt.

3. A középső rész: TabControl

  • Keress a Toolboxban egy TabControl-t és húzd be a Form közepére. Húzd szét nagyra.
  • Állítsd az Enabled tulajdonságát False-ra (így kezdetben inaktív lesz).
  • Nevezd el (Name): tabControl.
  • Kattints a jobb felső sarkában lévő kis nyílra, majd az Add Tab gombbal érd el, hogy 2 fül legyen:
    • 1. fül Text tulajdonsága: "Hozzávalók"
    • 2. fül Text tulajdonsága: "Elkészítés"

4. Hozzávalók fül berendezése (1. Tab)

  • Kattints a "Hozzávalók" fülre. Húzz be ide egy DataGridView-t.
    • (Name): dgvHozzavalok
    • AllowUserToAddRows: False
    • Kattints a jobb felső kis nyilára -> Edit Columns... Adj hozzá 3 oszlopot: "Mennyiség", "Egység", és "Megnevezés" (ennek a Width-jét tedd pl. 300-ra).
  • Húzz alá balra egy Button-t. (Name): btnHozzaad, Text: "Sor hozzáadása". Kattints duplán!
  • Húzz mellé még egy Button-t. (Name): btnTorol, Text: "Kijelölt sor törlése". Kattints duplán!

5. Elkészítés fül berendezése (2. Tab)

  • Kattints az "Elkészítés" fülre a tervezőben.
  • Húzz be egy TextBox-ot.
    • (Name): txtElkeszites
    • Multiline: True (nagyon fontos, különben 1 soros lesz!)
    • ScrollBars: Vertical
    • Húzd szét, hogy kitöltse a fület.

6. Mentés gomb

  • Húzz be a Form legalsó részére (a TabControl alá) egy Button-t.
  • (Name): btnMentes
  • Text: "Mentés"
  • Kattints duplán a gombra!
  • Kész is vagy! Ezzel minden esemény megvan a kódban. Most már csak be kell másolni a logikai változókat, függvényeket és az események kódjait a 9-es pontból a megfelelő helyekre.
