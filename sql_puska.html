1 SELECT lekérdezés teljes felépítése

A SELECT záradékok kötelező sorrendje
SELECT   mezo1, mezo2, AGG_FUGGVENY(mezo3) AS alias
FROM     tabla1
JOIN     tabla2 ON tabla1.id = tabla2.kulso_id   -- opcionális
WHERE    feltétel                                  -- sorok szűrése (csoportosítás ELŐTT)
GROUP BY mezo1, mezo2                              -- csoportosítás
HAVING   AGG_FUGGVENY(mezo3) > 5                  -- csoportok szűrése (csoportosítás UTÁN)
ORDER BY mezo1 ASC, mezo2 DESC                     -- rendezés
LIMIT    10;                                       -- hány sort adjunk vissza
A sorrendet nem lehet felcserélni — mindig ez a sorrend: SELECT → FROM → JOIN → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT

2 WHERE — sorok szűrése

Mikor? Ha feltétel alapján csak bizonyos sorokat szeretnénk látni.
Összehasonlító operátorok
SELECT nev, ar FROM termek WHERE ar > 1000;
SELECT nev FROM diak WHERE atlag >= 4.5;
SELECT cim FROM film WHERE ev <> 2000;   -- nem egyenlő
SELECT nev FROM ember WHERE kor BETWEEN 18 AND 65;
-- BETWEEN mindkét határt beleérti!
Logikai operátorok (AND, OR, NOT)
-- AND: mindkét feltétel teljesül
SELECT nev FROM termek WHERE ar > 100 AND ar < 500;

-- OR: valamelyik teljesül
SELECT nev FROM diak WHERE osztaly = 'A' OR osztaly = 'B';

-- NOT: tagadás
SELECT nev FROM emp WHERE NOT osztaly = 'HR';

-- Zárójelezés fontos!
WHERE (a = 1 OR b = 2) AND c = 3
IN — értéklista
-- IN: szerepel a felsorolt értékek között
SELECT nev FROM varos WHERE orszag IN ('HU', 'AT', 'SK');

-- NOT IN: egyik sem egyezik
SELECT nev FROM diak WHERE osztaly NOT IN ('A', 'B');

-- Egyenértékű:
WHERE osztaly = 'A' OR osztaly = 'B'
-- ezzel:
WHERE osztaly IN ('A', 'B')
Összetett feltételek kombinálása
-- Több feltétel AND-del:
SELECT cim FROM film
WHERE ev > 2000
  AND nyert > 0
  AND jatekido < 120;

-- Holtverseny esetén HAVING-gel is szűrünk:
SELECT nev, COUNT(*) AS db
FROM tabla
GROUP BY nev
HAVING COUNT(*) = (SELECT MAX(db) FROM (...) AS al);

3 Számított mező — AS alias

Mikor? Ha a SELECT-ben új értéket kell kiszámolni a meglévő mezőkből.
Aritmetikai kifejezések
-- Összeg, különbség, szorzat:
SELECT nev, (kesz + epul + terv) AS osszeg FROM munka;
SELECT nev, ar * 1000 / tomeg AS fajlag FROM anyag;
SELECT nev, (vegszam - kezdszam) AS terjedelem FROM adat;

-- Százalék:
SELECT nev, (teljesitett / osszes) * 100 AS szazalek FROM feladat;

-- AS = az új oszlop neve (idézőjel ha szóköz van):
SELECT ar * 1.27 AS "Bruttó ár" FROM termek;
ROUND, TRUNCATE, ABS
-- Kerekítés:
SELECT ROUND(ar / 3, 2) AS harmad FROM termek;
-- ROUND(szam, tizedesek)

-- Csonkítás (lefelé kerekít):
SELECT TRUNCATE(atlag, 1) FROM eredmeny;

-- Abszolút érték:
SELECT ABS(kulonbseg) AS elteres FROM mertek;

-- IF feltétel SELECT-ben:
SELECT nev,
  IF(ar > 1000, 'drága', 'olcsó') AS kategoria
FROM termek;

4 ORDER BY — rendezés

Mikor? Ha az eredményt sorba kell rendezni.
Rendezési variációk
-- Egyszerű, növekvő (alapértelmezett):
SELECT nev, ar FROM termek ORDER BY ar;
SELECT nev, ar FROM termek ORDER BY ar ASC;

-- Csökkenő:
SELECT nev, ar FROM termek ORDER BY ar DESC;

-- Több szempont (elsősorban nev, azon belül ar szerint):
SELECT nev, osztalyba, ar FROM tabla ORDER BY osztalyba, ar DESC;

-- Rendezés sorszámmal (2. oszlop szerint):
SELECT nev, COUNT(*) FROM tabla GROUP BY nev ORDER BY 2 DESC;

-- Számított mező szerint:
SELECT nev, ar * tomeg AS ertek FROM anyag ORDER BY ertek DESC;

5 DISTINCT — duplikátumok eltávolítása

Mikor? Ha ismétlődő értékek lehetnek, és minden értéket csak egyszer szeretnél látni.
Alap DISTINCT
-- Hány különböző rendező van?
SELECT DISTINCT rendezo FROM film;

-- Szűréssel kombinálva:
SELECT DISTINCT rendezo FROM film WHERE ev > 2000;

-- Több mezőre: a kombináció legyen egyedi
SELECT DISTINCT osztaly, varos FROM diak;
DISTINCT vs GROUP BY
-- Egyedi értékek lekérése — mindkettő működik:
SELECT DISTINCT nev FROM tabla;
-- ekvivalens:
SELECT nev FROM tabla GROUP BY nev;

-- Ha COUNT-ot is kell → GROUP BY kell:
SELECT nev, COUNT(*) FROM tabla GROUP BY nev;

-- DISTINCT COUNT-ban (egyedi nem-NULL értékek):
SELECT COUNT(DISTINCT varos) FROM diak;

6 JOIN — táblák összekapcsolása

Mikor? Ha az adatok több táblában vannak, és össze kell őket fűzni.
Implicit JOIN (régi stílus — WHERE-es)
-- Két tábla:
SELECT t1.mezo, t2.mezo
FROM tabla1 t1, tabla2 t2
WHERE t1.id = t2.kulso_id;

-- Három tábla (kapcsolótáblával):
SELECT f.cim, s.szinesz
FROM film f, kapcsolat k, szinesz s
WHERE f.id = k.film_id
  AND s.id = k.szinesz_id;
-- Mindkét idegen kulcsot be kell kötni!
Explicit JOIN (új stílus — JOIN ON)
-- Két tábla:
SELECT t1.mezo, t2.mezo
FROM tabla1 t1
JOIN tabla2 t2 ON t1.id = t2.kulso_id;

-- LEFT JOIN — bal tábla összes sora, jobb csak ha van:
SELECT t1.mezo, t2.mezo
FROM tabla1 t1
LEFT JOIN tabla2 t2 ON t1.id = t2.kulso_id;
-- Ha nincs egyezés, t2 mezői NULL-ok lesznek
Három+ tábla összekapcsolása
-- Implicit (leggyakoribb vizsgán):
SELECT a.nev, b.cim, c.ar
FROM tabla_a a, tabla_b b, tabla_c c
WHERE a.id = b.a_id
  AND b.id = c.b_id
  AND a.varos = 'Budapest';

-- Explicit:
SELECT a.nev, c.ar
FROM tabla_a a
JOIN tabla_b b ON a.id = b.a_id
JOIN tabla_c c ON b.id = c.b_id
WHERE a.varos = 'Budapest';
Szöveges mezőn alapuló kapcsolat
-- Ha nem id-n, hanem szöveges mezőn kötünk össze
-- (pl. rendező neve = színész neve):
SELECT DISTINCT rendezo AS szinesz_rendezo
FROM film, szinkron
WHERE film.rendezo = szinkron.szinesz;

-- Alias (AS) elhagyható FROM-ban:
SELECT f.cim FROM film f, szinkron s
WHERE f.filmaz = s.filmaz;
⚠️ Ha n darab táblát kapcsolsz össze, legalább n-1 feltételt kell írni a WHERE-be (vagy ON-ba). Kapcsolótáblánál mindkét idegen kulcsot be kell kötni!

7 GROUP BY + Aggregált függvények

Mikor? Ha csoportonként kell megszámolni, összegezni, max/min értéket keresni.
Aggregált függvények
COUNT(*)         -- sorok száma (NULL-t is számolja)
COUNT(mezo)      -- nem NULL értékek száma
COUNT(DISTINCT mezo) -- egyedi nem-NULL értékek
SUM(mezo)        -- összeg
AVG(mezo)        -- átlag
MAX(mezo)        -- legnagyobb érték
MIN(mezo)        -- legkisebb érték
MAX(ev)-MIN(ev)  -- terjedelem
GROUP BY minta
-- Előadónként hány dalt adott ki?
SELECT eloado, COUNT(*) AS dalszam
FROM zenek
GROUP BY eloado;

-- Tanulónként összesített jegy:
SELECT nev, AVG(jegy) AS atlag, COUNT(*) AS db
FROM jegyek
GROUP BY nev
ORDER BY atlag DESC;
Feladat szövege Aggregált függvény
hány ... van / darabszámCOUNT(*)
összesen / összegeSUM(mezo)
átlagos / átlagaAVG(mezo)
legtöbb / legnagyobb / legmagasabbMAX(mezo)
legkevesebb / legkisebb / legalacsonyabbMIN(mezo)
hány különbözőCOUNT(DISTINCT mezo)
⚠️ A SELECT-ben csak olyan mező szerepelhet, ami GROUP BY-ban van, VAGY aggregált függvénybe van csomagolva! Minden más mező → hibás lekérdezés.

8 HAVING — csoportszűrés

Mikor? Ha csoportosítás UTÁN kell szűrni — pl. "amelyik csoportban több mint 2 rekord van".

WHERE

Sorok szűrése csoportosítás ELŐTT
Nem használhat aggregált függvényt!

HAVING

Csoportok szűrése csoportosítás UTÁN
Használhat COUNT, SUM, MAX stb.-t!

HAVING alap minták
-- Legalább 10 filmmel rendelkező évek:
SELECT ev, COUNT(*) AS db
FROM film
GROUP BY ev
HAVING COUNT(*) >= 10;

-- Legalább 2 filmet rendező rendezők:
SELECT rendezo, COUNT(*) AS db
FROM film
GROUP BY rendezo
HAVING COUNT(*) > 1
ORDER BY db DESC;
WHERE + HAVING kombinálása
-- WHERE szűr a csoportosítás előtt,
-- HAVING szűr utána:
SELECT eloado, COUNT(*) AS db
FROM zenek
WHERE ev > 2000               -- csak 2000 utáni zenék
GROUP BY eloado
HAVING COUNT(*) >= 5          -- amelyiktől legalább 5 van
ORDER BY db DESC;

9 LIMIT — legjobb N elem

Mikor? Ha csak a legtöbbet / legnagyobb / legdrágábbat kell visszaadni.
LIMIT variációk
-- Legtöbb résztvevőjű esemény:
SELECT nev, COUNT(*) AS db
FROM esemeny, resztvevo
WHERE esemeny.id = resztvevo.esemeny_id
GROUP BY nev
ORDER BY db DESC
LIMIT 1;    -- csak az első (legtöbb)

-- LIMIT kihagyás, darab:
LIMIT 0, 5   -- első 5 sor
LIMIT 5, 5   -- 6-10. sor (kihagyja az első 5-öt)
LIMIT 1, 1   -- 2. sor (kihagyja az elsőt)
-- LIMIT kihagyás, darab
⚠️ Holtverseny — LIMIT nem elég!
-- Ha LIMIT 1-et használsz, holtverseny esetén
-- csak egyet ad vissza! Megoldás: skaláris allekérdezés

-- Mindazok, akiknek a legtöbb pontjuk van:
SELECT nev, SUM(pont) AS ossz
FROM pontok
GROUP BY nev
HAVING SUM(pont) = (
    SELECT MAX(osszeg) FROM (
        SELECT SUM(pont) AS osszeg
        FROM pontok
        GROUP BY nev
    ) AS al
);

10 IN / NOT IN — halmazos szűrés

Mikor? Ha egy halmaz alapján kell szűrni — "amelynek id-ja szerepel egy másik lekérdezés eredményében".
IN — benne van
-- Azok az elemek, amelyek teljesítenek egy másik feltételt
-- (pl. valamit tartalmaznak / kapcsolatban vannak valamivel):
SELECT nev FROM termek
WHERE id IN (
    SELECT termek_id FROM kapcsolat, tulajdonsag
    WHERE tulajdonsag.id = kapcsolat.tulaj_id
      AND tulajdonsag.nev = 'keresett_ertek'
);
NOT IN — nincs benne
-- Azok, amelyek NEM rendelkeznek valamivel:
SELECT nev FROM termek
WHERE id NOT IN (
    SELECT termek_id FROM kapcsolat, tulajdonsag
    WHERE tulajdonsag.id = kapcsolat.tulaj_id
      AND tulajdonsag.nev = 'kizarandó_ertek'
);
IN + IN — mindkét feltétel teljesül
-- Azok, amelyek EGYSZERRE rendelkeznek A-val ÉS B-vel:
SELECT nev FROM termek
WHERE id IN (
    SELECT termek_id FROM kapcsolat
    WHERE tulaj_id = (SELECT id FROM tulaj WHERE nev = 'A')
)
AND id IN (
    SELECT termek_id FROM kapcsolat
    WHERE tulaj_id = (SELECT id FROM tulaj WHERE nev = 'B')
);
IN + IN alternatíva HAVING-gel
-- Ugyanaz HAVING COUNT-tal (elegánsabb):
SELECT termek.nev
FROM termek, kapcsolat, tulaj
WHERE termek.id = kapcsolat.termek_id
  AND tulaj.id = kapcsolat.tulaj_id
  AND tulaj.nev IN ('A', 'B')         -- keresett értékek
GROUP BY termek.nev
HAVING COUNT(tulaj.id) = 2;           -- mindkettő megvan
⚠️ Miért nem elég WHERE tulaj.nev != 'X'? Mert ha egy terméknek van X és nem-X tulajdonsága is, a nem-X sor miatt a termék megjelenik. A NOT IN az egész terméket kizárja, ha bármelyik tulajdonsága X.

11 Skaláris allekérdezés WHERE/HAVING-ben

Mikor? Ha egyetlen értékkel kell összehasonlítani, amit egy másik lekérdezés ad vissza (pl. "akkora mint a maximum").
Maximum értéket elérők (holtverseny kezelve)
-- Mindazok, akiknek a legtöbb valami van:
SELECT nev, COUNT(*) AS db
FROM tabla1, tabla2
WHERE tabla1.id = tabla2.kulso_id
GROUP BY nev
HAVING COUNT(*) = (
    SELECT MAX(darab) FROM (
        SELECT COUNT(*) AS darab
        FROM tabla1, tabla2
        WHERE tabla1.id = tabla2.kulso_id
        GROUP BY tabla1.nev
    ) AS al    -- a FROM allekérdezésnek kötelező alias!
);
Egyszerűbb skaláris allekérdezés
-- Azok, akiknek az átlaguk feletti a pontjuk:
SELECT nev, pont
FROM diak
WHERE pont > (SELECT AVG(pont) FROM diak);

-- Azok, akiknek a max pontjuk van:
SELECT nev FROM diak
WHERE pont = (SELECT MAX(pont) FROM diak);

-- Legdrágábbhoz képest 80%-nál drágábbak:
SELECT nev, ar FROM termek
WHERE ar > (SELECT MAX(ar) FROM termek) * 0.8;

12 FROM allekérdezés — virtuális tábla

Mikor? Ha egy összetett eredményt tovább kell szűrni/feldolgozni, de egyetlen lekérdezésben nem megoldható.
Virtuális tábla FROM-ban
-- Összesítés után tovább szűrünk:
SELECT *
FROM (
    SELECT nev, SUM(ar) AS osszeg
    FROM rendeles
    GROUP BY nev
) AS osszesites             -- kötelező alias!
WHERE osszeg > 100000;

-- Több virtuális tábla összekötve:
SELECT ind.vonalid, ind.nev AS kezdo, veg.nev AS vegallomas
FROM (
    SELECT nev, vonalid FROM allomas, hely
    WHERE allomas.id = allomasid AND tav = 0
) AS ind,
(
    SELECT nev, vonalid, tav FROM allomas, hely
    WHERE allomas.id = allomasid
) AS veg,
(
    SELECT vonalid, MAX(tav) AS maxtav FROM hely GROUP BY vonalid
) AS tulso
WHERE ind.vonalid = veg.vonalid
  AND veg.vonalid = tulso.vonalid
  AND veg.tav = tulso.maxtav;
💡 A FROM-ban lévő allekérdezésnek mindig kell alias (AS nev). Nélküle szintaxis hiba!

13 Önillesztés — ugyanaz a tábla kétszer

Mikor? Ha ugyanabban a táblában kell két különböző sort összehasonlítani (pl. "aki X-szel együtt szerepelt").
Önillesztés minta
-- Akik 'XY'-jal együtt szerepeltek ugyanabban a filmben:
SELECT s1.szinesz, film.cim
FROM film, szinkron s1, szinkron s2
           -- ^^^^^^^^^^^^^^^^^^^^^ kétszer ugyanaz a tábla!
WHERE film.id = s1.film_id
  AND s1.film_id = s2.film_id  -- ugyanaz a film
  AND s2.szinesz = 'XY'        -- az egyik személy
  AND s1.szinesz <> 'XY'       -- de ne maga legyen
ORDER BY film.cim, s1.szinesz;
Mikor kell önillesztés?
-- Hierarchikus kapcsolat ugyanabban a táblában:
-- pl. ki a vezető (vezeto_id saját id-ra mutat)
SELECT dolgozo.nev, fonok.nev AS vezeto
FROM alkalmazott dolgozo, alkalmazott fonok
WHERE dolgozo.vezeto_id = fonok.id;

-- Hasonló ár/érték összehasonlítás:
SELECT t1.nev, t2.nev
FROM termek t1, termek t2
WHERE t1.ar = t2.ar
  AND t1.id < t2.id;  -- mindkét sorrend elkerülése

14 LIKE — mintaillesztés, LENGTH

LIKE minták (MySQL: % és _)
LIKE '%alma%'    -- tartalmazza az "alma" szót
LIKE 'B%'        -- B-vel kezdődik
LIKE '%s'        -- s-re végződik
LIKE 'B%s'       -- B-vel kezdődik, s-re végződik
LIKE '___'       -- pontosan 3 karakteres (_ = 1 karakter)
LIKE 'A__%'      -- A-val kezdődik, legalább 3 hosszú

-- Nem tartalmazza (NOT LIKE):
WHERE nev NOT LIKE '%törölt%'

-- MySQL: wildcard = %  (NOT Access = *)
LENGTH() — karakterszám
-- Pontosan 3 karakteres kód:
SELECT nev FROM termek WHERE LENGTH(kod) = 3;

-- Legalább 10 karakteres neve:
SELECT nev FROM allomas WHERE LENGTH(nev) >= 10;

-- LENGTH + LIKE kombinálva:
SELECT ut FROM europa
WHERE LENGTH(eurout) = 3 AND eurout LIKE 'E%5';

-- CHAR_LENGTH vs LENGTH:
-- LENGTH = bájtok száma (UTF8: ékezetes = 2 bájt!)
-- CHAR_LENGTH = karakterek száma (ékezetesnél helyes!)

15 Dátumfüggvények

Dátumból rész kinyerése
YEAR(datum)        -- évszám → pl. 2023
MONTH(datum)       -- hónap → 1-12
DAY(datum)         -- nap → 1-31
HOUR(idopont)      -- óra
MINUTE(idopont)    -- perc

-- Használat:
SELECT cim FROM film WHERE YEAR(bemutato) >= 2000;
SELECT cim FROM film
WHERE YEAR(bemutato) BETWEEN 1939 AND 1945;

-- Számítás dátumokkal:
SELECT cim FROM film
WHERE YEAR(bemutato) - gyartas_ev > 10;
Dátum egyéb függvények
DATEDIFF(d1, d2)  -- napok száma d1 és d2 között
CURDATE()         -- mai dátum
NOW()             -- aktuális dátum+idő

-- Életkor kiszámolása:
SELECT nev, YEAR(CURDATE()) - YEAR(szuldatum) AS kor
FROM szemely;

-- Legutóbbi X napon belüli rekordok:
SELECT * FROM rendeles
WHERE datum >= DATE_SUB(CURDATE(), INTERVAL 30 DAY);

16 NULL kezelés

IS NULL / IS NOT NULL
-- Hiányzó értékre szűrés:
SELECT nev, varos FROM szemely WHERE varos IS NULL;

-- Meg van adva az érték:
SELECT nev FROM allomas WHERE orszag IS NOT NULL;

-- SOHA ne használj = NULL-t!
WHERE mezo = NULL     -- MINDIG HAMIS! (nem működik)
WHERE mezo IS NULL    -- HELYES
IFNULL — NULL helyettesítése
-- Ha NULL, akkor 0-t jelenít meg:
SELECT nev, IFNULL(ar, 0) AS ar FROM termek;

-- Ha NULL, szöveget ad vissza:
SELECT nev, IFNULL(varos, 'ismeretlen') AS varos
FROM szemely;

-- COALESCE: az első nem-NULL értéket adja vissza:
SELECT COALESCE(email, telefon, 'nincs elérhetőség')
FROM kontakt;

17 Szövegfüggvények

Alap szövegfüggvények
UPPER(mezo)           -- nagybetűre konvertál
LOWER(mezo)           -- kisbetűre konvertál
TRIM(mezo)            -- szóközök levágása
LTRIM / RTRIM         -- bal/jobb oldali szóköz
LENGTH(mezo)          -- karakterszám (bájtban!)
CHAR_LENGTH(mezo)     -- karakterszám (ékezetesnél jobb)
CONCAT(a, ' ', b)     -- összefűzés
SUBSTRING(mezo, 1, 3) -- részstring (start, hossz)
REPLACE(mezo,'a','b') -- csere
INSTR(mezo, 'keres')  -- pozíció (0 ha nincs)
Szövegre példák
-- Teljes névből csak a vezetéknév (szóközig):
SELECT SUBSTRING(nev, 1, INSTR(nev, ' ')-1) AS vezeteknev
FROM szemely;

-- Nagy kezdőbetű ellenőrzés:
SELECT nev FROM varos
WHERE SUBSTRING(nev, 1, 1) = UPPER(SUBSTRING(nev, 1, 1));

-- Összefűzés:
SELECT CONCAT(vezeteknev, ' ', keresztnev) AS teljes_nev
FROM szemely;

18 Végrehajtási sorrend — hogyan dolgozza fel az SQL

A feldolgozás sorrendje (belső logika)
-- Az SQL NEM a leírás sorrendjében dolgozza fel!
-- Feldolgozási sorrend:
-- 1. FROM  (táblák kiválasztása)
-- 2. JOIN  (összekapcsolás)
-- 3. WHERE (sorok szűrése)
-- 4. GROUP BY (csoportosítás)
-- 5. HAVING (csoportok szűrése)
-- 6. SELECT (mezők kiválasztása, alias névadás)
-- 7. DISTINCT (duplikátumok eltávolítása)
-- 8. ORDER BY (rendezés)
-- 9. LIMIT (sorok korlátozása)
Ebből következő szabályok
-- ❌ WHERE-ben NEM lehet aggregált függvény:
WHERE COUNT(*) > 5      -- HIBA!
-- ✓ HAVING-ben kell:
HAVING COUNT(*) > 5     -- HELYES

-- ❌ WHERE-ben NEM lehet SELECT-ben definiált alias:
SELECT ar * 2 AS dupla
WHERE dupla > 100        -- HIBA! (még nem létezik)
-- ✓ Eredeti kifejezéssel:
WHERE ar * 2 > 100       -- HELYES

-- ✓ ORDER BY-ban viszont LEHET alias:
SELECT ar * 2 AS dupla ...
ORDER BY dupla DESC      -- HELYES

19 Tipikus feladatminták — kész sablonok

🏆 Legtöbb X-szel rendelkező elem (holtverseny kezelve)
-- "Melyik terméknek van a legtöbb összetevője?"
SELECT tabla1.nev, COUNT(*) AS db
FROM tabla1, kapcsolat, tabla2
WHERE tabla1.id = kapcsolat.id1 AND tabla2.id = kapcsolat.id2
GROUP BY tabla1.nev
HAVING COUNT(*) = (
    SELECT MAX(darab) FROM (
        SELECT COUNT(*) AS darab
        FROM tabla1, kapcsolat, tabla2
        WHERE tabla1.id = kapcsolat.id1 AND tabla2.id = kapcsolat.id2
        GROUP BY tabla1.id
    ) AS al
);
🔍 Olyan elemek, amelyek EGYSZERRE rendelkeznek A-val és B-vel
-- "Paradicsomos ÉS chilis keverékek" — HAVING COUNT = 2
SELECT tabla1.nev
FROM tabla1, kapcsolat, tabla2
WHERE tabla1.id = kapcsolat.id1 AND tabla2.id = kapcsolat.id2
  AND tabla2.nev IN ('A', 'B')
GROUP BY tabla1.nev
HAVING COUNT(tabla2.id) = 2;

-- Alternatíva IN + IN-nel:
SELECT nev FROM tabla1
WHERE id IN (SELECT id1 FROM kapcsolat, tabla2
             WHERE tabla2.id = kapcsolat.id2 AND tabla2.nev = 'A')
AND   id IN (SELECT id1 FROM kapcsolat, tabla2
             WHERE tabla2.id = kapcsolat.id2 AND tabla2.nev = 'B');
🚫 Olyan elemek, amelyek NEM tartalmaznak valamit
-- "Amelyek NEM tartalmaznak borsot"
SELECT nev FROM tabla1
WHERE id NOT IN (
    SELECT id1 FROM kapcsolat, tabla2
    WHERE tabla2.id = kapcsolat.id2
      AND tabla2.nev LIKE '%bors%'
);
📊 Csoportonkénti statisztika szűréssel
-- "Melyik kategóriában volt legalább 5 eladás, átlagban 10.000 Ft felett?"
SELECT kategoria, COUNT(*) AS db, AVG(ar) AS atlag
FROM termek
JOIN eladas ON termek.id = eladas.termek_id
WHERE eladas.datum >= '2023-01-01'       -- csak 2023-as eladások
GROUP BY kategoria
HAVING COUNT(*) >= 5                     -- legalább 5 eladás
   AND AVG(ar) > 10000                   -- átlag feletti
ORDER BY atlag DESC;
👥 Önillesztés — "akik együtt szerepeltek"
-- "Akik XY-jal EGYÜTT szerepeltek (de nem XY maga)"
SELECT s1.szemely, esemeny.nev
FROM esemeny, szereples s1, szereples s2
WHERE esemeny.id = s1.esemeny_id
  AND s1.esemeny_id = s2.esemeny_id
  AND s2.szemely = 'XY'
  AND s1.szemely <> 'XY'
ORDER BY esemeny.nev, s1.szemely;
📈 Rangsorolás — top N, 2. helyezett
-- Legtöbb eladással rendelkező termék (1. hely):
SELECT nev, COUNT(*) AS db
FROM termek, eladas WHERE termek.id = eladas.termek_id
GROUP BY nev ORDER BY db DESC LIMIT 1;

-- 2. legtöbbet eladott (1 kihagyás):
SELECT nev, COUNT(*) AS db
FROM termek, eladas WHERE termek.id = eladas.termek_id
GROUP BY nev ORDER BY db DESC LIMIT 1, 1;
--                                   ^ kihagyás, ^ db
